var calendar1 = new calendar({
appendUnder: "section",//指定父节点
format: "d/m/y",//指定显示格式
display: false,//初始是否显示ture || false
viewMode: 0,//初始赋值，0，1，2（日，月，年）
Multiselect: {//日期多选模式配置
available: true,//日期多选可用
minNumDays: 3,
maxNumDays: 100
}
});
calendar1.init();

}
